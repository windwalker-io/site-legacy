WINDWALKER

A PHP Rapid Application Development Framework

Start your application with a simple and understandable interface. 

namespace Flower\Controller\Sakura;

use Windwalker\Core\Controller\AbstractController;

/**
* Single action controller to handle GET request.
*/
class GetController extends AbstractController
{
    public function doExecute()
    {
        $page = $this->input->getInt('page', 1);

        $sakuras = $this->getRepository()->getSakuras($page, 15);

        return $this->renderView('Sakura', 'default', 'edge', compact('sakuras'));
    }
}
